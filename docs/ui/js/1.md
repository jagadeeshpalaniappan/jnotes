## 2. Remove duplicates from Array (OneLiner)

```js
let inputNums = [1,2,2,3];

// Array to Set
let set = new Set(inputNums);
console.log(set); // 1,2,3

// Set to Array:
console.log([...set]); // [1,2,3]

// OneLiner:
console.log([...new Set(inputNums)]); // [1,2,3]
```



## 3. let vs var 

```js
let myFn = function() {
  {
    let l = 'let';
    var v = 'var';
  }
  
  console.log(v);
  console.log(l);
}
```

Make 'v' as block scope, (condn: do not change 'var')

```js
let myFn = function() {

  (function() {
  
    {
      let l = 'let';
      var v = 'var';
    }
    
  })();

  console.log(v);
  console.log(l);
}

```
